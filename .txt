model ProductAttribute {
    id        BigInt @id @default(autoincrement())

    code      String @unique @db.VarChar(100)
    name      String @db.VarChar(150)
    dataType  ProductAttributeDataType @map("data_type")
    unit      String? @db.VarChar(50)

    @@map("product_attributes")
}

enum ProductAttributeDataType {
    TEXT
    NUMBER
    BOOLEAN
    ENUM
    JSON
}

    model ProductAttributeValue {
    id          BigInt @id @default(autoincrement())

    productId   BigInt @map("product_id")
    attributeId BigInt @map("attribute_id")

    valueText    String?  @map("value_text") @db.Text
    valueNumber  Decimal? @map("value_number") @db.Decimal(14, 4)
    valueBoolean Boolean? @map("value_boolean")
    valueJson    Json?    @map("value_json")

    product   Product          @relation(fields: [productId], references: [id], onDelete: Cascade)
    attribute ProductAttribute @relation(fields: [attributeId], references: [id], onDelete: Cascade)

    @@unique([productId, attributeId])
    @@map("product_attribute_values")
}


model Organization {
    id          BigInt @id @default(autoincrement())

    name        String @unique @db.VarChar(255)
    slug        String @unique @db.VarChar(255)

    description String? @db.Text
    email       String? @db.VarChar(255)
    phone       String? @db.VarChar(80)

    isBrand        Boolean @default(false) @map("is_brand")
    isManufacturer Boolean @default(false) @map("is_manufacturer")
    isSupplier     Boolean @default(false) @map("is_supplier")

    brandedProducts      Product[] @relation("ProductBrand")
    manufacturedProducts Product[] @relation("ProductManufacturer")
    suppliedProducts     Product[] @relation("ProductSupplier")

    createdAt DateTime @default(now()) @map("created_at")
    updatedAt DateTime @updatedAt @map("updated_at")

    @@map("organizations")
}

model ProductType {
    id          BigInt @id @default(autoincrement())
    code        String @unique @db.VarChar(100)
    name        String @db.VarChar(150)
    description String? @db.Text

    categoryId  BigInt? @map("category_id")
    category    ProductCategory? @relation(fields: [categoryId], references: [id])

    products    Product[]

    @@map("product_types")
}

model Product {
    id               BigInt @id @default(autoincrement())

    sku              String @unique @db.VarChar(255)
    slug             String @unique @db.VarChar(255)

    name             String @db.VarChar(255)
    displayName      String @map("display_name") @db.VarChar(255)

    description      String? @db.Text
    descriptionSeo   String? @map("description_seo") @db.VarChar(160)
    titleSeo         String? @map("title_seo") @db.VarChar(70)

    brandId          BigInt? @map("brand_id")
    manufacturerId   BigInt? @map("manufacturer_id")

    categoryId       BigInt? @map("category_id")
    productTypeId    BigInt? @map("product_type_id")

    basePriceAmount  Decimal? @map("base_price_amount") @db.Decimal(12, 3)
    vatRate          Decimal  @default(19.000) @map("vat_rate") @db.Decimal(5, 3)

    stockQuantity    Decimal @default(0) @map("stock_quantity") @db.Decimal(12, 3)
    stockUnit        ProductStockUnit @default(ITEM) @map("stock_unit")

    visibility       Boolean @default(true)
    priceVisibility  Boolean @default(true) @map("price_visibility")
    stockVisibility  Boolean @default(false) @map("stock_visibility")

    commercialMode   ProductCommercialMode @default(ON_REQUEST_OR_ONLINE) @map("commercial_mode")

    guaranteeMonths  Int @default(0) @map("guarantee_months")

    supplierId       BigInt? @map("supplier_id")
    createdById      BigInt? @map("created_by_id")
    updatedById      BigInt? @map("updated_by_id")

    datasheetMediaId BigInt? @map("datasheet_media_id")

    createdAt        DateTime @default(now()) @map("created_at")
    updatedAt        DateTime @updatedAt @map("updated_at")

    brand            Organization? @relation("ProductBrand", fields: [brandId], references: [id])
    manufacturer     Organization? @relation("ProductManufacturer", fields: [manufacturerId], references: [id])
    supplier         Organization? @relation("ProductSupplier", fields: [supplierId], references: [id])

    category         ProductCategory? @relation(fields: [categoryId], references: [id])
    productType      ProductType? @relation(fields: [productTypeId], references: [id])

    laptop           Laptop?

    @@map("products")
}
